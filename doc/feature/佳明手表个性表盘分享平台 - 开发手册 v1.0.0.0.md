## 五、数据库设计规范

### 5.1 表设计原则

- 主键：使用 uuid 或自增 serial
- 时间戳：必须有 created_at，可选 updated_at
- 软删除：使用 deleted_at 而非物理删除
- 外键：明确 onDelete 行为（cascade / set null / restrict）

### 6.2 API Route 模板

**运动记录接口：**

```TypeScript
// src/app/api/activities/route.ts
import { NextRequest, NextResponse } from "next/server";
import { db } from "@/lib/db";
import { activities } from "@/lib/db/schema";
import { auth } from "@/lib/auth";
import { desc, eq } from "drizzle-orm";

// GET /api/activities
export async function GET(request: NextRequest) {
  try {
    const session = await auth();
    if (!session?.user) {
      return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
    }

    const searchParams = request.nextUrl.searchParams;
    const page = parseInt(searchParams.get("page") || "1");
    const limit = parseInt(searchParams.get("limit") || "20");

    const data = await db.query.activities.findMany({
      where: eq(activities.userId, session.user.id),
      limit,
      offset: (page - 1) * limit,
      orderBy: [desc(activities.startTime)],
    });

    return NextResponse.json({ 
      activities: data,
      pagination: { page, limit }
    });
  } catch (error) {
    console.error("API Error:", error);
    return NextResponse.json(
      { error: "Internal Server Error" },
      { status: 500 }
    );
  }
}
```

## 七、Garmin API 集成

### 7.1 Garmin OAuth 流程

```TypeScript
// src/lib/garmin/oauth.ts
import { env } from "@/lib/env";

const GARMIN_AUTH_URL = "https://connect.garmin.com/oauthConfirm";
const GARMIN_TOKEN_URL = "https://apis.garmin.com/oauth/oauth/token";

export function generateGarminAuthUrl(state: string): string {
  const params = new URLSearchParams({
    client_id: env.GARMIN_CLIENT_ID!,
    response_type: "code",
    redirect_uri: `${env.NEXT_PUBLIC_APP_URL}/api/auth/garmin/callback`,
    state,
  });

  return `${GARMIN_AUTH_URL}?${params.toString()}`;
}

export async function exchangeCodeForToken(code: string) {
  const response = await fetch(GARMIN_TOKEN_URL, {
    method: "POST",
    headers: {
      "Content-Type": "application/x-www-form-urlencoded",
    },
    body: new URLSearchParams({
      grant_type: "authorization_code",
      code,
      client_id: env.GARMIN_CLIENT_ID!,
      client_secret: env.GARMIN_CLIENT_SECRET!,
      redirect_uri: `${env.NEXT_PUBLIC_APP_URL}/api/auth/garmin/callback`,
    }),
  });

  if (!response.ok) {
    throw new Error("Failed to exchange code for token");
  }

  return response.json();
}
```

### 8.1 Canvas 渲染工具

```TypeScript
// src/lib/poster/canvas.ts
import { Activity } from "@/lib/db/schema";

export interface PosterConfig {
  width: number;
  height: number;
  template: "achievement" | "minimal" | "art" | "trail";
  data: {
    raceName: string;
    finishTime: string;
    avgPace: string;
    distance: string;
    avgHr: number;
    elevationGain: number;
    date: string;
    customText?: string;
  };
  style: {
    primaryColor: string;
    fontFamily: string;
  };
}

export function formatActivityData(activity: Activity): PosterConfig["data"] {
  const hours = Math.floor(activity.durationSeconds! / 3600);
  const minutes = Math.floor((activity.durationSeconds! % 3600) / 60);
  const seconds = activity.durationSeconds! % 60;
  
  const finishTime = `${hours.toString().padStart(2, "0")}:${minutes.toString().padStart(2, "0")}:${seconds.toString().padStart(2, "0")}`;
  
  const paceMinutes = Math.floor(activity.avgPaceSeconds! / 60);
  const paceSeconds = activity.avgPaceSeconds! % 60;
  const avgPace = `${paceMinutes}:${paceSeconds.toString().padStart(2, "0")} /km`;
  
  const distance = `${(activity.distanceMeters! / 1000).toFixed(1)} km`;
  
  return {
    raceName: activity.name || "跑步训练",
    finishTime,
    avgPace,
    distance,
    avgHr: activity.avgHr || 0,
    elevationGain: activity.elevationGain || 0,
    date: formatDate(activity.startTime!),
  };
}
```

### 8.2 路线图渲染

```TypeScript
// src/lib/poster/route-renderer.ts
import polyline from "polyline";

interface Point {
  lat: number;
  lng: number;
}

export function decodePolyline(encoded: string): Point[] {
  return polyline.decode(encoded).map(([lat, lng]) => ({ lat, lng }));
}

export function renderRoute(
  ctx: CanvasRenderingContext2D,
  encodedPolyline: string,
  options: {
    canvasWidth: number;
    canvasHeight: number;
    padding: number;
    strokeColor: string;
    strokeWidth: number;
  }
): void {
  const points = decodePolyline(encodedPolyline);
  if (points.length < 2) return;
  
  const bounds = calculateBounds(points);
  const { canvasWidth, canvasHeight, padding, strokeColor, strokeWidth } = options;
  
  // 计算缩放比例和偏移
  const latRange = bounds.maxLat - bounds.minLat;
  const lngRange = bounds.maxLng - bounds.minLng;
  const scale = Math.min(
    (canvasWidth - padding * 2) / lngRange,
    (canvasHeight - padding * 2) / latRange
  );
  
  // 绘制路线
  ctx.beginPath();
  ctx.strokeStyle = strokeColor;
  ctx.lineWidth = strokeWidth;
  
  points.forEach((point, i) => {
    const x = (point.lng - bounds.minLng) * scale + padding;
    const y = canvasHeight - (point.lat - bounds.minLat) * scale - padding;
    
    if (i === 0) ctx.moveTo(x, y);
    else ctx.lineTo(x, y);
  });
  
  ctx.stroke();
}
```
